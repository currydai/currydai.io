四字真言：增删改查

1.线性表

  • 定义：n个具有相同类型的数据元素的有限序列
  • 逻辑特征：个数的有限性、类型的相通性，类型的抽象性、序偶关系

1.1顺序表

  • 定义：线性表的顺序存储结构
  • 要点：用一段连续地址，依次存储线性表中的数据元素
#pragma once
#include <iostream>
using namespace std;

const int MaxSize = 100;

template<class T>

class SeqList
{
public:
	SeqList();
	SeqList(T a[], int n);
	//~SeqList();

	//增
	T Insert(int i, T x);

	//删
	T Delete(int i);

	//改
	//好像不能直接修改

	//查
	int Empty();
	int GetLength();
	T Get(int i);
	int GetIndex(T x);

	//打印
	void Print();

private:
	int length;
	T data[MaxSize];
};

//类外实现

//初始化。建立一个空表
template<class T>
SeqList<T>::SeqList()
{
	length = 0;
}

//构造函数，建立表
template<class T>
SeqList<T>::SeqList(T a[], int n)
{
	if (n > MaxSize)
	{
		throw "参数非法";
	}

	for (int i = 0; i < n; i++)
	{
		data[i] = a[i];
	}

	length = n;
}


//////////增
template<class T>
T SeqList<T>::Insert(int i, T x)
{
	if (length == MaxSize)
	{
		throw "上溢,无法再插入！";
	}

	if (i<1 || i>length)
	{
		throw "插入位置错误！";
	}

	//i后面多有元素后移
	for (int j = length; j >= i; j--)
	{
		data[j] = data[j - 1];
	}

	//把x的值放在i-1处
	data[i - 1]=x; 

	//数组长度+1
	length++;
}

///////删
template<class T>
T SeqList<T>::Delete(int i)
{
	if (i<0 || i>length)
	{
		throw "删除位置错误！";
	}

	for (int j = i; j < length; j++)
	{
		data[j-1] = data[j];
	}

	length--;
}

//////查
template<class T>
int SeqList<T>::Empty()
{
	if (length == 0)
	{
		return 1;
	}

	else
	{
		return 0;
	}
}

template<class T>
int SeqList<T>::GetLength()
{
	return length;
}

template<class T>
T SeqList<T>::Get(int i)
{
	if (i<0 || i>length)
	{
		throw "查询位置非法！";
	}
	else
	{
		return data[i];
	}
}

template<class T>
int SeqList<T>::GetIndex(T x)
{
	int flag = 0;
	int index;
	for (int j = 0; j < length; j++)
	{
		if (x == data[j])
		{
			index=j;
			flag++;
			break;
		}
	}

	if (flag > 0)
	{
		return index;
	}

	else
	{
		throw "无法查到到此元素！";
	}

}

template<class T>
void SeqList<T>::Print()
{
	for (int j = 0; j < length; j++)
	{
		cout << data[j];
	}
}




#include<iostream>
using namespace std;
#include "SeqList.h"

int main()
{
	int c[5] = {1,2,3,4,5};
	SeqList<int> arr1(c, 5);

	arr1.Insert(2, 0);
	arr1.Print();
	cout << "----------------" << endl;

	arr1.Delete(2);
	arr1.Print();
	cout << "----------------" << endl;

	int ans=arr1.Get(2);
	cout << ans << endl;
	cout << "----------------" << endl;

	int ans2=arr1.GetIndex(2);
	cout << ans2 << endl;
	cout << "----------------" << endl;


	return 0;
}

1.2 单链表

  • 顺序表->静态存储分配->事先确定容量
  • 单链表->动态存储分配->运行时分配空间
  • 单链表：线性表的链式存储结构
  • 思想：用一组任意的存储单元存放线性表
  • 特性：逻辑次序和物理次序不一定相同，元素之间的逻辑关系用指针表示
  • 节点结构：data->数据域 next指针域
  • 特殊性：头指针->指向第一个节点的存储地址，尾标志->终端节点的指针域为空 头结点->在第一个元素节点之前附设一个类型相同的结点
  • 头指针：一个指向第一个节点地址的指针变量
  • 头指针具有标识单链表的作用，所以经常用头指针代表单链表的名字
  • 头结点：在单链表的第一个结点之前附设一个结点，它没有直接前驱，称之为头结点
  • 可不存信息，也可以作为监视哨，或用于存放线性表的长度等附加信息
  • 指针域中存放首元结点的地址
  • 首元结点：存储第一个元素的节点
    -链表如果为空的情况下，如果单链表没有头结点，那么头指针就会指向NULL，如果加上头结点，无论单链表是否为空，头指针都会指向头结点，这样使得空链表与非空链表处理一致。并且使首元结点前插入或删除元素的时候，与后面操作相同，不需要产生额外的判断分支，使得算法更加简单。
