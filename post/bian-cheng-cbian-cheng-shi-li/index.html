基础知识

C++中的数据类型：

  • 1字节：char
  • 2字节：short
  • 4字节：int float long
  • 8字节：double longlong
    此外，4中整型都有一种不能储存负数值的无符号变体（unsigned），主要优点是增大变量的储存最大值。

重载的<< and >>运算符：

.txt格式文件读取

#include<iostream>
#include<fstream>
#include<string>
using namespace std;

int main(int argc,char *argv[])
{

	cout << "argc" << ":"<<argc << endl;
	cout << "*argv" << ":" << *argv << endl;
    //文件写操作
	ofstream fout("test.txt");

	string a = "abc";
	int b[2][2];
	for (int i = 0; i < 2; i++)
	{
		for (int j = 0; j < 2; j++)
		{
			b[i][j] = i + j;
			fout << b[i][j];
			fout << " ";
		}
		fout << '\n';
	}
	fout.close();
	cout << "文件保存结束" << endl;


	//文件读操作
	ifstream fin("test.txt");
	//char c;
	//fin >> c;


	string d;
	while (getline(fin, d))
	{
		cout << d << endl;
	}

	cout << "文件读取结束" << endl;
	fin.close();

	return 0;
}
